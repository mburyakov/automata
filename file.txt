class manual() {
update ->  This.
}
class base(Arg:manual) {
update() ->
    the_method(Arg).
the_method(A) ->
    Arg + A.
}

class linear(Arg1:manual, Arg2:manual) extends base(Arg1), manual() {
update() ->
    manual:update(),
    base:the_method(Arg2).
}
